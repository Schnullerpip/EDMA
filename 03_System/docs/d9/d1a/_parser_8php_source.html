82  // nach s:\t beliebige Anzahl von Buchstaben \p{L}, Ziffern \p{N}, Punkt,

90  // Nummer Pattern

354  private function checkUhrzeitFormat($zeit) {
355  $retVal = array();
356  $zeit_splitted = preg_split("/[\.,]/", $zeit);
357  $zeiten_teile = count($zeit_splitted);
358  if ($zeiten_teile < 2) {
359  $hh_mm_ss = explode(":", $zeit);
360  if ($hh_mm_ss[2] > 59) { // Fehler, wenn Sekunden groesser als 59 (Mikrosekunden ohne '.' oder ',' angehaengt
361  $this->throwMessException("Ungültiges Uhrzeitformat in Zeile " .
362  ($j + $this->_zeilennummerMessdaten) . ": " . $zeit);
363  }
364  $uhrzeit = $zeit;
365  $mikrosekunden = "000000";
366  } else if ($zeiten_teile > 2) {
367  $this->throwMessException("Ungültiges Uhrzeitformat in Zeile " .
368  ($j + $this->_zeilennummerMessdaten) . ": " . $zeit);
369  } else {
370  $uhrzeit = $zeit_splitted[0];
371  $mikrosekunden = $zeit_splitted[1];
372  if ($mikrosekunden > 999999) { // Fehler, wenn Mikrosekunden aus mehr als 6 Stellen bestehen
373  $this->throwMessException("Sekunden haben mehr als sechs Nachkommastellen in Zeile " .
374  ($j + $this->_zeilennummerMessdaten) . ": " . $zeit);
375  }
376  $mikrosekunden = $this->fillMikroseks($mikrosekunden);
377  }
378  $retVal["uhrzeit"] = $uhrzeit;
379  $retVal["mikroseks"] = $mikrosekunden;
380  return $retVal;
